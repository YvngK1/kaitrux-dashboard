// PrefixCommands/info/status.js const { EmbedBuilder } = require('discord.js'); const os = require('os'); module.exports = { name: 'status', aliases: ['botstatus', 'ping', 'uptime'], description: 'Muestra el estado del bot, base de datos y Lavalink.', category: 'info', run: async (client, message, args) => { const mongoose = require('mongoose'); // Medir latencia del mensaje const sent = await message.channel.send('📡 Calculando...'); const apiPing = sent.createdTimestamp - message.createdTimestamp; await sent.delete().catch(() => {}); // Uptime formateado const uptimeSecs = Math.floor(process.uptime()); const d = Math.floor(uptimeSecs / 86400); const h = Math.floor((uptimeSecs % 86400) / 3600); const m = Math.floor((uptimeSecs % 3600) / 60); const s = uptimeSecs % 60; const uptimeStr = [d && `${d}d`, h && `${h}h`, m && `${m}m`, `${s}s`].filter(Boolean).join(' '); // DB const dbState = mongoose.connection.readyState; const dbEmoji = dbState === 1 ? '🟢' : dbState === 2 ? '🟡' : '🔴'; const dbLabel = ['❌ Desconectado', '✅ Conectado', '🔄 Conectando', '⏳ Desconectando'][dbState] || '❓ Desconocido'; // Lavalink let lavalinkStr = '❓ Desconocido'; if (client.poru && client.poru.nodes) { const nodes = []; client.poru.nodes.forEach((node, name) => { const icon = node.isConnected ? '🟢' : '🔴'; const playing = node.stats?.playingPlayers ?? '?'; nodes.push(`${icon} \`${name}\` — ${playing} playing`); }); lavalinkStr = nodes.length ? nodes.join('\n') : '⚪ Sin nodos'; } // RAM const usedMB = Math.round((process.memoryUsage().heapUsed) / 1024 / 1024); const totalMB = Math.round(os.totalmem() / 1024 / 1024); // Servidores / usuarios const guilds = client.guilds.cache.size; const users = client.guilds.cache.reduce((acc, g) => acc + g.memberCount, 0); const embed = new EmbedBuilder() .setTitle('📊 Estado de Kaitrux') .setColor(0x5865F2) .setThumbnail(client.user.displayAvatarURL()) .addFields( { name: '🤖 Bot', value: [ `**Estado:** 🟢 Online`, `**Ping WS:** \`${client.ws.ping}ms\``, `**Ping API:** \`${apiPing}ms\``, `**Uptime:** \`${uptimeStr}\``, ].join('\n'), inline: true, }, { name: '📊 Estadísticas', value: [ `**Servidores:** \`${guilds}\``, `**Usuarios:** \`${users.toLocaleString()}\``, `**Comandos:** \`${client.commands.size}\``, `**RAM:** \`${usedMB}MB\``, ].join('\n'), inline: true, }, { name: '🗄️ Base de datos', value: `${dbEmoji} ${dbLabel}`, inline: false, }, { name: '🎵 Lavalink', value: lavalinkStr, inline: false, }, ) .setFooter({ text: `Node.js ${process.version} • Kaitrux`, iconURL: client.user.displayAvatarURL() }) .setTimestamp(); message.channel.send({ embeds: [embed] }); }, };
